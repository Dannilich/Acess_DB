 SELECT студент.Фамилия & " " & Left(студент.Имя,1) & "." & Left(студент.Отчество,1) & "." AS ФИО, Avg(Ведомость.Отметка) AS [Средний балл]
FROM студент INNER JOIN Ведомость ON студент.Код = Ведомость.[Код студента]
GROUP BY студент.Код, студент.Фамилия, студент.Имя, студент.Отчество;


SELECT [Страховые случаи].Случай, Count([Страховые договоры].Код) AS [Count-Код]
FROM [Страховые случаи] INNER JOIN [Страховые договоры] ON [Страховые случаи].Код = [Страховые договоры].Случай
GROUP BY [Страховые случаи].Случай;

SELECT Автор.Фамилия & " " & Left(Автор.Имя,1) & "." & Left(Автор.Отчество,1) & "." AS [Автор], COUNT([Книга Автор].[Код книги]) AS [Количество книг]
FROM Автор INNER JOIN [Книга Автор] ON Автор.Код = [Книга Автор].[Код автора]
GROUP BY Автор.Код, Автор.Фамилия, Автор.Имя, Автор.Отчество;
